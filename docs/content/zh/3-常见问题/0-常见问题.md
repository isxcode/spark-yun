> runAsUser: 1000 需要和zhiqingyun的uid一致

```yml
apiVersion: v1
kind: Pod
metadata:
  name: host-pod
spec:
  containers:
    - volumeMounts:
      - mountPath: /etc/passwd
        name: users-volume
  restartPolicy: never
  securityContext:
    privileged: true
    runAsUser: 1000  
  hostAliases:
    - ip: "172.16.215.83"
      hostnames:
        - "isxcode"
  volumes:
    - name: users-volume
      hostPath:
        path: /etc/passwd
```
